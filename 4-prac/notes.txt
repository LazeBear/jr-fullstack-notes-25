folder structure

/--root level ---/
-- node_modules/
-- package.json
-- package-lock.json
-- src/
  |-- index.js 入口文件 (app.js , server.js)
  |-- routes/
    |-- movie.router.js (movieRouter.js, movie-router.js, movies.router.js)
    |-- user.router.js
  |-- controllers/ (validation, business logic)
    |-- movie.controller.js
    |-- user.controller.js
  |-- models/ (DB CRUD operation)
    |-- user.model.js
  |-- middleware
    |-- error/
      |-- error.middleware.js
    |-- cors.middleware.js
    |-- authGuard.middleware.js
  |-- config/
  |-- utils/ (utility function, helper function)
    |-- db.js (db connection)
    |-- config.js
